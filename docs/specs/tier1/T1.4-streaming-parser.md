## Current Problem

`_read_file_content()` at line 412 reads entire file into memory. Then `content.strip().split('\n')` is called 3 times (detect, validate, parse). Memory = ~3x file size. Two 50MB VCF files = 700MB peak memory.

## Implementation

### 1. Add `_line_iterator()` function

```python
import io

def _line_iterator(file):
    """Yield lines from file without loading entire content into memory."""
    if isinstance(file, (str, Path)):
        with open(Path(file), encoding='utf-8') as handle:
            for line in handle:
                yield line.rstrip('\n\r')
    elif isinstance(file, BytesIO):
        file.seek(0)
        wrapper = io.TextIOWrapper(file, encoding='utf-8')
        for line in wrapper:
            yield line.rstrip('\n\r')
        wrapper.detach()  # Don't close the underlying BytesIO
    else:
        for line in file:
            line_str = line if isinstance(line, str) else line.decode('utf-8')
            yield line_str.rstrip('\n\r')
```

### 2. Add `_read_header_lines()` function

```python
def _read_header_lines(file, max_lines=50):
    """Read first N lines for format detection without loading entire file."""
    lines = []
    for i, line in enumerate(_line_iterator(file)):
        lines.append(line)
        if i >= max_lines - 1:
            break
    # Reset file position for subsequent full parse
    if hasattr(file, 'seek'):
        file.seek(0)
    return lines
```

### 3. Add file size validation

```python
MAX_FILE_SIZE_BYTES = 200 * 1024 * 1024  # 200 MB
MAX_SNP_COUNT = 10_000_000

def _validate_file_size(file):
    """Validate file size before parsing."""
    if isinstance(file, (str, Path)):
        size = Path(file).stat().st_size
    elif isinstance(file, BytesIO):
        size = len(file.getvalue())
    else:
        return  # Can't check size for generic file objects

    if size > MAX_FILE_SIZE_BYTES:
        raise ValueError(f"File too large ({size / 1024 / 1024:.1f} MB). Maximum: {MAX_FILE_SIZE_BYTES / 1024 / 1024:.0f} MB")
```

### 4. Refactor `parse_genetic_file()` to use two-phase approach

```python
def parse_genetic_file(file):
    _validate_file_size(file)

    # Phase 1: Detect format from header lines only
    header_lines = _read_header_lines(file, max_lines=50)
    detected_format = _detect_format_from_lines(header_lines)

    # Phase 2: Stream-parse the full file
    if hasattr(file, 'seek'):
        file.seek(0)

    parser_map = {
        "23andMe": _parse_23andme_streaming,
        "AncestryDNA": _parse_ancestry_streaming,
        "MyHeritage": _parse_myheritage_streaming,
        "VCF": _parse_vcf_streaming,
    }

    parser = parser_map.get(detected_format)
    if not parser:
        raise ValueError(f"Unsupported format: {detected_format}")

    return parser(file), detected_format
```
